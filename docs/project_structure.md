# Структура проекта

```bash
erudition-app # корень проекта
├── api       # контракты api (OpenAPI, gRPC)
│   └── proto # protobuf-схемы для gRPC
├── docs      # документация
├── pkg       # общий код, который используют другие микросервисы
├── secrets   # чувсвительные данные (данные для захода в бд и пр.)
└── services  # микросервисы
    ├── knowledge
    │   ├── cmd/main.go # входная точка программы
    │   └── internal    # приватный код программы
    │       ├── adapters    # адаптеры внешних систем (бд, внешние api)
    │       ├── config      # загрузка конфигурации (.env)
    │       └── core        # бизнес-ядро (не знает об инфраструктуре)
    │           ├── models      # модели предметной области
    │           ├── ports       # интерфейсы, которые адаптеры реализуют
    │           └── services    # бизнес-логика, опирации над моделями
    ├── questions
    └── и пр.
```
